import nltk
nltk.download('punkt')
nltk.download('stopwords')
nltk.download('wordnet')
nltk.download('omw-1.4')


from nltk.corpus import wordnet as wn


s1="The Natural Language Toolkit (NLTK) is an open-source Python library for NLP"
#s1='This is a sample example'
s1_token=s1.split()
synset_coll=[]

for i in s1_token:
  temp=wn.synsets(i)
  if (len(temp)>0):
      print("\n",temp,len(temp))
      synset_coll.append(temp)
for i in synset_coll:
  print(i) 
  print("\n")   

print("\n")
print("Name                     Definition                         Example")
for i in synset_coll:
  for j in range(len(i)):
    print(i[j].name(), '   -  ',i[j].definition(),'  -  ',i[j].examples())
    #print(synset_coll[0][1].lemmas()[1].name())

#printing Hypernym and Hyponym of the words in a sentence
print("Name              Hypernyms                     Hyponyms                     Root-Hypernyms")
for i in synset_coll:
  for j in range(len(i)):
    print(i[j].name(),'  -  ',i[j].hypernyms(),'  -  ',i[j].hyponyms(),'  -  ',i[j].root_hypernyms())



print("Name                   Pos")
for i in synset_coll:
  for j in range(len(i)):
    print(i[j].name(),'    -   ',i[j].pos())
